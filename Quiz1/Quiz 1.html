  • Print list_emissions

# create lists - china and korea
list_dataset_countries=["data_quiz/china_co2_emissions.csv","data_quiz/korea_rep_co2_emissions.csv"]
list_emissions=[]

# for loop
for i in range(0,2):
    dataset=pd.read_csv(list_dataset_countries[i])
    total_emissions_per_country=np.array(dataset["total_emissions"])
    list_emissions.append(round(np.mean(total_emissions_per_country),3))

print(list_emissions) # china and korea means espectively
